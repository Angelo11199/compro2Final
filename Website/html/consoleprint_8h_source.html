94std::string centerText(const std::string& text, int maxWidth = 0, char sideChar = ' ') {
95 maxWidth = getWidth();
96 int totalPadding = maxWidth - text.size() - 2; // subtract 2 for the side characters
97 if (totalPadding < 0) totalPadding = 0; // prevent negative padding
98 int leftPadding = totalPadding / 2;
99 int rightPadding = totalPadding - leftPadding; // in case totalPadding is odd
100 return sideChar + std::string(leftPadding, ' ') + text + std::string(rightPadding, ' ') + sideChar;
