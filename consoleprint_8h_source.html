86std::string centerText(const std::string& text, int maxWidth = 1, char sideChar = ' ') {
87 int totalPadding = maxWidth - text.size() - 2; // subtract 2 for the side characters
88 if (totalPadding < 0) totalPadding = 0; // prevent negative padding
89 int leftPadding = totalPadding / 2;
90 int rightPadding = totalPadding - leftPadding; // in case totalPadding is odd
91 return sideChar + std::string(leftPadding, ' ') + text + std::string(rightPadding, ' ') + sideChar;
